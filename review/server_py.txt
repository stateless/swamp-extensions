#!/usr/bin/env python3
"""Generic human-in-the-loop review canvas (Python stdlib only).

Serves ONE review session described by a spec JSON file:

  list mode — a curation grid: items grouped by `group`, each with a
              configurable option-scale (radios), an optional context
              select, and a comment field. For gating long LLM-generated
              candidate lists (migrations, triage, cleanup approvals).
  doc mode  — a pastebin/textedit canvas: markdown editor on the left,
              live preview on the right, plus an optional verdict
              option-scale (e.g. approve/revise/reject) for publish gates.
              With multiple files, a directory listing renders above the
              editor — per-file verdict + comment, so the listing doubles
              as a checklist over the files.

Every Save dual-writes into the spec's `outDir`:
  <name>.json    — structured result (what `collect` reads)
  <name>.md      — human-readable render (doc mode: the edited doc itself)
  <name>.log.md  — timestamped append-log (a save can never silently lose
                   prior state — history survives)

Auth: every request must carry `?t=<token>`. Intended ONLY for secure
private networks; the token is an abuse guard, not internet-grade auth.

Usage:
  python3 server.py --spec spec.json --bind 0.0.0.0 --port 8800 --token T
"""

P = argparse.ArgumentParser()
P.add_argument("--spec", required=True)
P.add_argument("--bind", default="0.0.0.0")
P.add_argument("--port", type=int, default=8800)
P.add_argument("--token", required=True)
ARGS = P.parse_args()

with open(ARGS.spec) as f:
    SPEC = json.load(f)

NAME = SPEC["name"]
MODE = SPEC.get("mode", "list")
TITLE = SPEC.get("title", NAME)
OUT = SPEC["outDir"]
SCALE = SPEC.get("optionScale") or (["keep", "decide", "drop"] if MODE == "list" else [])
CONTEXTS = SPEC.get("contexts") or []
ITEMS = SPEC.get("items") or []
CONTENT = SPEC.get("content") or ""
INSTRUCTIONS = SPEC.get("instructions") or ""

# doc mode is internally always multi-file; bare content degenerates to one.
FILES = SPEC.get("files") or []
if MODE == "doc" and not FILES:
    FILES = [{"name": "doc", "title": TITLE, "content": CONTENT}]

CONTENT_VERSION = 1  # bumped on each reload_spec(); polled by the page for the source-changed banner

SAVE_JSON = os.path.join(OUT, NAME + ".json")
SAVE_MD = os.path.join(OUT, NAME + ".md")
SAVE_LOG = os.path.join(OUT, NAME + ".log.md")
SERVER_LOG = os.path.join(OUT, ".review.%s.server.log" % NAME)


def slog(msg):
    ts = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    try:
        with open(SERVER_LOG, "a") as f:
            f.write("%s %s\n" % (ts, msg))
    except OSError:
        pass


def load_saved():
    try:
        with open(SAVE_JSON) as f:
            return json.load(f)
    except (OSError, ValueError):
        return {}


def reload_spec():
    """Re-read the spec file from disk and rebind state, in place — the standard
    unix service 'reload config' convention (SIGHUP, or POST /reload for runtimes
    that can't signal). No restart, so the port/token and any open tab survive;
    the browser just needs a refresh to render the new content."""
    global SPEC, NAME, MODE, TITLE, OUT, SCALE, CONTEXTS, ITEMS, CONTENT
    global INSTRUCTIONS, FILES, SAVE_JSON, SAVE_MD, SAVE_LOG, SERVER_LOG, CONTENT_VERSION
    try:
        with open(ARGS.spec) as f:
            spec = json.load(f)
    except (OSError, ValueError) as e:
        slog("reload failed (spec unreadable): %s" % e)
        return False
    if not os.path.isabs(spec.get("outDir", "")):
        slog("reload refused: outDir %r is not absolute (would bury saves "
             "under the server CWD)" % spec.get("outDir"))
        return False
    SPEC = spec
    NAME = SPEC["name"]
    MODE = SPEC.get("mode", "list")
    TITLE = SPEC.get("title", NAME)
    OUT = SPEC["outDir"]
    SCALE = SPEC.get("optionScale") or (["keep", "decide", "drop"] if MODE == "list" else [])
    CONTEXTS = SPEC.get("contexts") or []
    ITEMS = SPEC.get("items") or []
    CONTENT = SPEC.get("content") or ""
    INSTRUCTIONS = SPEC.get("instructions") or ""
    FILES = SPEC.get("files") or []
    if MODE == "doc" and not FILES:
        FILES = [{"name": "doc", "title": TITLE, "content": CONTENT}]
    SAVE_JSON = os.path.join(OUT, NAME + ".json")
    SAVE_MD = os.path.join(OUT, NAME + ".md")
    SAVE_LOG = os.path.join(OUT, NAME + ".log.md")
    SERVER_LOG = os.path.join(OUT, ".review.%s.server.log" % NAME)
    CONTENT_VERSION += 1
    slog("reloaded spec v%d (mode=%s, %d files)" % (CONTENT_VERSION, MODE, len(FILES)))
    return True

def scale_css():
    rules = []
    for i, v in enumerate(SCALE):
        color = KNOWN_COLORS.get(v, PALETTE[i % len(PALETTE)])
        rules.append(".r.s%d{color:%s}" % (i, color))
    return "\n".join(rules)

# Minimal client-side markdown renderer — headings, fences, lists, quotes,
# tables, hr, inline code/bold/italic/links. Preview-grade, not spec-complete.
MD_JS = """
function esc(s){return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;')}
function critic(s){
 s=s.replace(/\\{\\+\\+([\\s\\S]+?)\\+\\+\\}/g,'<ins>$1</ins>');
 s=s.replace(/\\{--([\\s\\S]+?)--\\}/g,'<del>$1</del>');
 s=s.replace(/\\{~~([\\s\\S]+?)~&gt;([\\s\\S]+?)~~\\}/g,'<del>$1</del><ins>$2</ins>');
 s=s.replace(/\\{==([\\s\\S]+?)==\\}/g,'<mark>$1</mark>');
 s=s.replace(/\\{&gt;&gt;([\\s\\S]+?)&lt;&lt;\\}/g,'<span class="critc">$1</span>');
 return s;
}
function inline(s){
 s=esc(s);
 s=critic(s);
 s=s.replace(/`([^`]+)`/g,function(m,c){return '<code>'+c+'</code>'});
 s=s.replace(/\\*\\*([^*]+)\\*\\*/g,'<b>$1</b>');
 s=s.replace(/(^|[^*])\\*([^*]+)\\*/g,'$1<i>$2</i>');
 s=s.replace(/~~([^~]+)~~/g,'<s>$1</s>');
 s=s.replace(/==([^=]+)==/g,'<mark>$1</mark>');
 s=s.replace(/\\[([^\\]]+)\\]\\(([^)\\s]+)\\)/g,'<a href="$2" target="_blank" rel="noopener">$1</a>');
 return s;
}
// Block-walker: returns [{l: startLine, h: html}] — one entry per top-level
// block. data-line stamps + per-block DOM patching hang off this.
function mdBlocks(src,isTop){
 var out=[],L=src.split('\\n'),i=0;
 function push(start,html){out.push({l:start,h:html})}
 if(isTop&&L[0]==='---'){
  var fj=1;while(fj<L.length&&L[fj]!=='---'&&L[fj]!=='...')fj++;
  if(fj<L.length){
   push(0,'<div class="fm"><div class="fmh">frontmatter</div><pre>'+esc(L.slice(1,fj).join('\\n'))+'</pre></div>');
   i=fj+1;
  }
 }
 while(i<L.length){
  var start=i,l=L[i];
  if(/^```/.test(l)){var b=[];i++;while(i<L.length&&!/^```/.test(L[i])){b.push(L[i]);i++}
   i++;push(start,'<pre><code>'+esc(b.join('\\n'))+'</code></pre>');continue}
  var h=l.match(/^(#{1,6})\\s+(.*)/);
  if(h){push(start,'<h'+h[1].length+'>'+inline(h[2])+'</h'+h[1].length+'>');i++;continue}
  if(/^\\s*(-{3,}|\\*{3,})\\s*$/.test(l)){push(start,'<hr>');i++;continue}
  if(/^\\s*>/.test(l)){var q=[];while(i<L.length&&/^\\s*>/.test(L[i])){q.push(L[i].replace(/^\\s*>\\s?/,''));i++}
   push(start,'<blockquote>'+mdRender(q.join('\\n'))+'</blockquote>');continue}
  if(/^\\s*([-*+]|\\d+\\.)\\s+/.test(l)){
   var ord=/^\\s*\\d+\\./.test(l),li=[];
   while(i<L.length&&/^\\s*([-*+]|\\d+\\.)\\s+/.test(L[i])){li.push('<li>'+inline(L[i].replace(/^\\s*([-*+]|\\d+\\.)\\s+/,''))+'</li>');i++}
   push(start,(ord?'<ol>':'<ul>')+li.join('')+(ord?'</ol>':'</ul>'));continue}
  if(/^\\s*\\|/.test(l)){var rows=[];
   while(i<L.length&&/^\\s*\\|/.test(L[i])){rows.push(L[i]);i++}
   var t='<table>';
   for(var r=0;r<rows.length;r++){
    if(/^\\s*\\|[\\s:|-]+\\|?\\s*$/.test(rows[r]))continue;
    var cells=rows[r].replace(/^\\s*\\||\\|\\s*$/g,'').split('|');
    var tag=(r===0)?'th':'td';
    t+='<tr>'+cells.map(function(c){return '<'+tag+'>'+inline(c.trim())+'</'+tag+'>'}).join('')+'</tr>';
   }
   push(start,t+'</table>');continue}
  if(/^\\s*$/.test(l)){i++;continue}
  var p=[];while(i<L.length&&!/^\\s*$/.test(L[i])&&!/^(#|```|\\s*>|\\s*([-*+]|\\d+\\.)\\s|\\s*\\|)/.test(L[i])){p.push(L[i]);i++}
  push(start,'<p>'+inline(p.join(' '))+'</p>');
 }
 return out;
}
function mdRender(src,isTop){
 return mdBlocks(src,isTop).map(function(b){return b.h}).join('\\n');
}
"""

def render_list_rows(saved):
    by_name = {it.get("name", ""): it for it in saved.get("items", [])}
    out, last = [], None
    for i, it in enumerate(ITEMS):
        name = it.get("name", "")
        group = it.get("group", "")
        meta = it.get("meta", "")
        note = it.get("note", "")
        sv = by_name.get(name)
        disp = (sv or {}).get("disposition") or it.get("recommendation") or ""
        if disp not in SCALE:
            disp = SCALE[0] if SCALE else ""
        cmt = (sv or {}).get("comment", "")
        ctx = (sv or {}).get("context") or it.get("context") or (CONTEXTS[0] if CONTEXTS else "")
        if group != last:
            out.append("<h2>%s</h2>" % html.escape(group))
            last = group
        radios = ""
        for si, v in enumerate(SCALE):
            chk = " checked" if v == disp else ""
            radios += (
                '<label class="r s%d"><input type="radio" name="d%d" value="%s"%s>%s</label>'
                % (si, i, html.escape(v, quote=True), chk, html.escape(v))
            )
        ctxsel = ""
        if CONTEXTS:
            opts = "".join(
                "<option%s>%s</option>" % (" selected" if c == ctx else "", html.escape(c))
                for c in CONTEXTS
            )
            ctxsel = '<select class="ctx" id="x%d">%s</select>' % (i, opts)
        note_s = (" &mdash; <i>%s</i>" % html.escape(note)) if note else ""
        out.append(
            '<div class="row" data-i="%d" data-group="%s" data-name="%s" data-meta="%s">'
            '<div class="meta"><b>%s</b> <span class="o">%s</span>%s</div>'
            '<div class="ctl"><span class="radios">%s</span>%s'
            '<input class="cm" id="c%d" value="%s" placeholder="comment"></div></div>'
            % (
                i, html.escape(group, quote=True), html.escape(name, quote=True),
                html.escape(meta, quote=True), html.escape(name), html.escape(meta),
                note_s, radios, ctxsel, i, html.escape(cmt, quote=True),
            )
        )
    return "\n".join(out)


def page_list():
    saved = load_saved()
    legend = INSTRUCTIONS or ("scale: " + " &middot; ".join("<b>%s</b>" % html.escape(v) for v in SCALE))
    return (
        "<!doctype html><html><head><meta charset=utf-8><title>%s</title>"
        "<style>%s\n%s</style></head><body>"
        "<header><h1>%s</h1>"
        "<textarea id=notes rows=2 placeholder='general notes / decisions...'>%s</textarea></header>"
        "<main>%s</main>"
        "<div id=bar><button onclick=save()>Save</button><span id=msg>%s</span></div>"
        "<script>%s%s</script></body></html>"
        % (
            html.escape(TITLE), CSS, scale_css(), html.escape(TITLE),
            html.escape(saved.get("notes", "")), render_list_rows(saved),
            legend, common_js(), LIST_JS,
        )
    )


def render_list_md(data):
    rows = data.get("items", [])
    notes = data.get("notes", "")
    has_ctx = bool(CONTEXTS)
    L = ["# %s" % TITLE, ""]
    if notes.strip():
        L += ["**Notes:** %s" % notes, ""]
    hdr = "| item | group | meta |" + (" context |" if has_ctx else "") + " disposition | comment |"
    sep = "|---|---|---|" + ("---|" if has_ctx else "") + "---|---|"
    L += [hdr, sep]
    for a in rows:
        cells = [a.get("name", ""), a.get("group", ""), a.get("meta", "")]
        if has_ctx:
            cells.append(a.get("context", ""))
        cells += ["**%s**" % a.get("disposition", ""), a.get("comment", "")]
        L.append("| " + " | ".join(cells) + " |")
    L += ["", "## By disposition"]
    for v in SCALE:

def file_md_path(fname):
    return os.path.join(OUT, "%s.%s.md" % (NAME, fname))


def sidecar_targets():
    """name -> absolute sidecar path (foo.webcanvas.md), from the loaded spec.
    Only present when serve_docs built the spec with per-file `sidecar` paths;
    absent for plain specs (so this is a no-op for them)."""
    out = {}
    for f in (SPEC.get("files") or []):
        if f.get("sidecar"):
            out[f.get("name", "")] = f["sidecar"]
    if SPEC.get("sidecar") and SPEC.get("content") is not None:
        out["__single__"] = SPEC["sidecar"]
    return out


def _write_side(path, content):
    body = content if (content.endswith("\n") or not content) else content + "\n"
    with open(path, "w") as f:
        f.write(body)


def write_sidecars(data):
    """Write each edited doc to its co-located <source>.webcanvas.md sidecar.
    The source file is never touched; the sidecar is what the webcanvas edits.
    Returns the number of sidecars written."""
    targets = sidecar_targets()
    if not targets:
        return 0
    files = data.get("files", [])
    if "__single__" in targets:  # single-doc spec — content may arrive in files[0] or top-level
        content = (files[0].get("content", "") if files else data.get("content", "")) or ""
        _write_side(targets["__single__"], content)
        return 1
    n = 0
    for fl in files:
        side = targets.get(fl.get("name", ""))
        if side:
            _write_side(side, fl.get("content", "") or "")
            n += 1
    return n


def render_doc_index_md(data):
    """Multi-file index: a table of files + verdicts; contents live per-file."""
    files = data.get("files", [])
    notes = data.get("notes", "")
    L = ["# %s" % TITLE, ""]
    if (data.get("verdict") or "").strip():
        L += ["**Verdict:** %s" % data["verdict"], ""]
    if notes.strip():
        L += ["**Notes:** %s" % notes, ""]
    L += ["| file | verdict | comment | bytes |", "|---|---|---|---|"]
    for f in files:
        content = f.get("content", "") or ""
        L.append("| [%s](%s) | **%s** | %s | %d |" % (
            f.get("title") or f.get("name", ""),
            os.path.basename(file_md_path(f.get("name", ""))),
            f.get("verdict", ""), f.get("comment", ""), len(content.encode()),
        ))
    return "\n".join(L) + "\n"


def log_doc(ts, data):
    files = data.get("files", [])
    if not files and data.get("content") is not None:
        files = [{"name": "doc", "content": data.get("content", "")}]
    notes = (data.get("notes", "") or "").strip()
    verdict = (data.get("verdict", "") or "").strip()
    log = ["", "---", "## %s  (%d files)" % (ts, len(files))]
    if verdict:
        log.append("**Verdict:** %s" % verdict)
    if notes:
        log.append("**Notes:** %s" % notes)
    for f in files:
        content = f.get("content", "") or ""
        digest = hashlib.sha256(content.encode()).hexdigest()[:12]
        entry = "- **%s**: %d bytes, %d lines, sha %s" % (
            f.get("name", ""), len(content.encode()),
            content.count("\n") + 1, digest,
        )
        if f.get("verdict"):
            entry += " — %s" % f["verdict"]
        if (f.get("comment") or "").strip():
            entry += " (%s)" % f["comment"]
        log.append(entry)
    return log

class Handler(http.server.BaseHTTPRequestHandler):
    def log_message(self, *a):
        pass

    def _authed(self):
        q = urllib.parse.urlparse(self.path).query
        t = urllib.parse.parse_qs(q).get("t", [""])[0]
        return hmac.compare_digest(t, ARGS.token)

    def _send(self, code, ctype, body):
        self.send_response(code)
        self.send_header("Content-Type", ctype)
        self.send_header("Content-Length", str(len(body)))
        self.end_headers()
        self.wfile.write(body)

    def do_GET(self):
        if not self._authed():
            return self._send(403, "text/plain", b"forbidden (missing/bad token)")
        path = urllib.parse.urlparse(self.path).path
        if path == "/ping":
            return self._send(200, "application/json",
                              json.dumps({"ok": True, "name": NAME, "mode": MODE,
                                          "version": CONTENT_VERSION}).encode())
        if path == "/state.json":
            return self._send(200, "application/json", json.dumps(load_saved()).encode())
        body = (page_list() if MODE == "list" else page_doc()).encode()
        return self._send(200, "text/html; charset=utf-8", body)

    def do_POST(self):
        if not self._authed():
            return self._send(403, "text/plain", b"forbidden (missing/bad token)")
        path = urllib.parse.urlparse(self.path).path
        if path == "/shutdown":
            # token-authenticated self-shutdown — the controller's stop/restart
            # path; no pid signalling needed (works from sandboxed runtimes).
            slog("shutdown requested")
            self._send(200, "application/json", b'{"ok": true, "stopping": true}')
            threading.Thread(target=SRV.shutdown, daemon=True).start()
            return
        if path == "/reload":
            # token-authenticated config reload (SIGHUP equivalent for sandboxed
            # runtimes that can't deliver signals). Re-reads the spec in place.
            ok = reload_spec()
            return self._send(200, "application/json",
                              json.dumps({"ok": ok, "reloaded": ok}).encode())
        if path == "/autosave":
            # lightweight client autosave: persist editor content to the
            # <source>.webcanvas.md sidecars only — no json/md/log churn.
            ln = int(self.headers.get("Content-Length", 0))
            try:
                data = json.loads(self.rfile.read(ln) or b"{}")
            except ValueError:
                return self._send(400, "text/plain", b"bad json")
            n = write_sidecars(data) if MODE == "doc" else 0
            ts = datetime.datetime.now().strftime("%H:%M:%S")
            return self._send(200, "application/json",
                              json.dumps({"ok": True, "savedAt": ts, "wrote": n}).encode())
        ln = int(self.headers.get("Content-Length", 0))
        try:
            data = json.loads(self.rfile.read(ln) or b"{}")
        except ValueError:
            return self._send(400, "text/plain", b"bad json")
        ts = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        data["savedAt"] = ts
        data["mode"] = MODE
        data["name"] = NAME
        os.makedirs(OUT, exist_ok=True)
        with open(SAVE_JSON, "w") as f:
            json.dump(data, f, indent=2)
        if MODE == "list":
            md, log = render_list_md(data), log_list(ts, data)
        else:
            files = data.get("files", [])
            if len(files) > 1:
                md = render_doc_index_md(data)
                for fl in files:
                    content = fl.get("content", "") or ""
                    with open(file_md_path(fl.get("name", "")), "w") as f:
                        f.write(
                            content if content.endswith("\n") or not content
                            else content + "\n"
                        )
            else:
                md = (files[0].get("content", "") if files else data.get("content", "")) or ""
            log = log_doc(ts, data)
            write_sidecars(data)  # also persist edits to the co-located sidecars
        with open(SAVE_MD, "w") as f:
            f.write(md if md.endswith("\n") or not md else md + "\n")
        with open(SAVE_LOG, "a") as f:
            f.write("\n".join(log) + "\n")
        slog("save: %d bytes json" % ln)
        r = json.dumps({"ok": True, "savedAt": ts, "path": SAVE_MD}).encode()
        return self._send(200, "application/json", r)


SRV = None


def main():
    global SRV
    if not os.path.isabs(OUT):
        # Loud on stderr too: slog can't write (its log dir is under the bad
        # OUT, which doesn't exist), so a deliberate refusal must not be silent.
        msg = ("FATAL: outDir %r is not absolute — refusing to start. A "
               "relative outDir resolves against the server CWD and silently "
               "buries saves (the tmp/ split-brain). Pass an absolute path."
               % OUT)
        print(msg, file=sys.stderr)
        slog(msg)
        sys.exit(2)
    os.makedirs(OUT, exist_ok=True)
    # Unix service convention: SIGHUP reloads the spec in place (no restart).
    try:
        signal.signal(signal.SIGHUP, lambda *_: reload_spec())
    except (AttributeError, ValueError):
        pass  # no SIGHUP (e.g. Windows / non-main thread) — /reload endpoint covers it
    socketserver.TCPServer.allow_reuse_address = True
    with socketserver.TCPServer((ARGS.bind, ARGS.port), Handler) as s:
        SRV = s
        slog("serving %s mode=%s on %s:%d" % (NAME, MODE, ARGS.bind, ARGS.port))
        print("serving on %s:%d" % (ARGS.bind, ARGS.port))
        s.serve_forever()
        slog("stopped")


if __name__ == "__main__":
    try:
        main()
    except Exception as e:  # noqa: BLE001 — last-resort breadcrumb for the detached process
        slog("fatal: %r" % e)
        sys.exit(1)
